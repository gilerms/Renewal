// Channel Warper :: channel
//============================================================
function	script	F_IzludeChannel	{
	mes "[Oficial do Canal]";
	mes "Izlude é um lugar muito animado, não é?";
	mes "Estou aqui para ajudar	no que precisar!";
	mes "Ou posso mandar você para a outra Izlude.";
	mes "Você quer ir para outra Izlude?";
	next;
	set .@i, select(getarg(0)+":Deixa pra lá");
	if (.@i < 6) {
		setarray .@maps$[1],"izlude","izlude_a","izlude_b","izlude_c","izlude_d";
		savepoint .@maps$[.@i],128,98,1,1;
		warp .@maps$[.@i],128,98;
		end;
	}
	close;
}
